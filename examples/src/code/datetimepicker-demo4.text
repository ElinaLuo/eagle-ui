import React,{Component,PropTypes} from 'react';
import {Input, DateTimePicker} from 'eagle-ui';


this.state = {
    dateTime:''
};

<DateTimePicker defaultValue={this.state.dateTime} getValueCallback={(val)=>{
    this.setState({
        dateTime:val
    });
}}>
    <Input placeholder="请选择日期时间"  icon="calendar" />
</DateTimePicker>
